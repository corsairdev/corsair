export const WindowsEnvVarsNote = () => (
  <Note>
    **Windows (PowerShell):** Commands that use `export VAR=value` or `$VAR` syntax are for macOS and Linux. In PowerShell, set environment variables with `$env:VAR_NAME = "value"` and reference them with `$env:VAR_NAME` — for example, `$env:GITHUB_TOKEN = "ghp_..."` and `api_key=$env:GITHUB_TOKEN`.
  </Note>
