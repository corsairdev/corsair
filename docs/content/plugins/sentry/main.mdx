## Quick Start

Install the plugin:

```bash
pnpm install @corsair-dev/sentry
```

Add the Sentry plugin to your Corsair instance:

```ts title="corsair.ts"
import { createCorsair } from "corsair";
import { sentry } from "@corsair-dev/sentry";

export const corsair = createCorsair({
    plugins: [
        sentry(),
    ],
});
```

Once configured, you can manage Sentry issues and projects:

```ts
// List issues for a project
const issues = await corsair.sentry.api.issues.list({
    organizationSlug: "my-org",
    projectSlug: "my-project",
});
```

## Authentication

### Supported Auth Types

The Sentry plugin supports:
- **`api_key`** (default) - Use a Sentry auth token

### Default Auth Type

If no `authType` is specified, the plugin defaults to **`api_key`**.

### Configuring API Key Authentication

Store credentials with the CLI:

```bash
pnpm corsair setup --sentry api_key=your-auth-token
```

For webhook signature verification:

```bash
pnpm corsair setup --sentry webhook_signature=your-webhook-secret
```

Alternatively, provide credentials directly in the config:

```ts title="corsair.ts"
sentry({
    key: process.env.SENTRY_AUTH_TOKEN,
    webhookSecret: process.env.SENTRY_WEBHOOK_SECRET,
})
```

See [Authentication](/concepts/auth) for details on managing credentials.

## Options

| Option | Type | Description |
|--------|------|-------------|
| `authType` | `'api_key'` | Authentication method (defaults to `'api_key'`) |
| `key` | `string` | Auth token (optional, uses database if not provided) |
| `webhookSecret` | `string` | Webhook secret for `sentry-hook-signature` verification |
| `hooks` | `object` | Endpoint hooks for custom logic |
| `webhookHooks` | `object` | Webhook hooks for event handling |
| `errorHandlers` | `object` | Custom error handlers |
| `permissions` | `object` | Permission configuration for AI agent access |

### Hooks

```ts
sentry({
    hooks: {
        issues: {
            update: {
                before: async (ctx, input) => {
                    console.log("Updating issue:", input.issueId);
                    return { ctx, input };
                },
            },
        },
    },
})
```

See [Hooks](/concepts/hooks) for complete documentation.

### Error Handling

The plugin includes built-in error handlers for common scenarios. For complete documentation, see the [Error Handlers reference](/plugins/sentry/error-handlers).

## Usage

### Accessing the API

```ts
// Get a specific issue
const issue = await corsair.sentry.api.issues.get({
    organizationSlug: "my-org",
    issueId: "ISSUE123",
});

// Resolve an issue
await corsair.sentry.api.issues.update({
    organizationSlug: "my-org",
    issueId: "ISSUE123",
    status: "resolved",
});

// Create a release
await corsair.sentry.api.releases.create({
    organizationSlug: "my-org",
    version: "v1.2.3",
    projects: ["my-project"],
});

// List teams
const teams = await corsair.sentry.api.teams.list({
    organizationSlug: "my-org",
});
```

See [API Endpoints](/plugins/sentry/api-endpoints) for the complete reference.

### Webhooks

Sentry sends issue and alert events (identified by `sentry-hook-signature` + `sentry-hook-resource` headers):

```ts title="app/api/webhook/route.ts"
import { processWebhook } from "corsair";
import { corsair } from "@/server/corsair";

export async function POST(request: Request) {
    const headers = Object.fromEntries(request.headers);
    const body = await request.json();

    const result = await processWebhook(corsair, headers, body);
    return result.response;
}
```

See [Webhooks](/plugins/sentry/webhooks) for all available events.

### Database Access

```ts
const issues = await corsair.sentry.db.issues.search({
    data: { status: "unresolved" },
});
```

See [Database](/plugins/sentry/database) for the complete schema.

### Multi-Tenancy

```ts
const tenant = corsair.withTenant("org-123");

const issues = await tenant.sentry.api.issues.list({
    organizationSlug: "my-org",
    projectSlug: "my-project",
    query: "is:unresolved",
});
```

## Examples

### Example 1: Auto-create Linear Issue on Sentry Error

```ts title="corsair.ts"
import { inngest } from "./inngest";

export const corsair = createCorsair({
    plugins: [
        sentry({
            webhookHooks: {
                issues: {
                    created: {
                        after: async (ctx, result) => {
                            await inngest.send({ // [!code highlight]
                                name: "sentry/issue-created", // [!code highlight]
                                data: { // [!code highlight]
                                    tenantId: ctx.tenantId, // [!code highlight]
                                    issueId: result.data.data?.issue?.id, // [!code highlight]
                                    title: result.data.data?.issue?.title, // [!code highlight]
                                    url: result.data.data?.issue?.web_url, // [!code highlight]
                                }, // [!code highlight]
                            }); // [!code highlight]
                        },
                    },
                },
            },
        }),
        linear(),
    ],
});
```

```ts title="inngest/functions.ts"
export const createLinearIssue = inngest.createFunction(
    { id: "create-linear-from-sentry" },
    { event: "sentry/issue-created" },
    async ({ event }) => {
        const tenant = corsair.withTenant(event.data.tenantId);

        await tenant.linear.api.issues.create({
            title: `[Sentry] ${event.data.title}`,
            teamId: process.env.LINEAR_TEAM_ID!,
            description: `Sentry issue: ${event.data.url}`, // [!code highlight]
            labelIds: ["bug-label-id"],
        });
    }
);
```

### Example 2: Notify on Metric Alert

```ts title="corsair.ts"
export const corsair = createCorsair({
    plugins: [
        sentry({
            webhookHooks: {
                alerts: {
                    metricAlert: {
                        after: async (ctx, result) => {
                            await inngest.send({
                                name: "sentry/metric-alert",
                                data: {
                                    tenantId: ctx.tenantId,
                                    alertTitle: result.data.data?.metric_alert?.title, // [!code highlight]
                                    status: result.data.data?.metric_alert?.status,
                                },
                            });
                        },
                    },
                },
            },
        }),
        slack(),
    ],
});
```

```ts title="inngest/functions.ts"
export const notifyMetricAlert = inngest.createFunction(
    { id: "notify-sentry-metric-alert" },
    { event: "sentry/metric-alert" },
    async ({ event }) => {
        const tenant = corsair.withTenant(event.data.tenantId);

        if (event.data.status === "critical") {
            await tenant.slack.api.messages.post({
                channel: "C_ALERTS",
                text: `🚨 Sentry metric alert: ${event.data.alertTitle}`, // [!code highlight]
            });
        }
    }
);
```
