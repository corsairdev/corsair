## Quick Start

Install the plugin:

```bash
pnpm install @corsair-dev/cal
```

Add the Cal.com plugin to your Corsair instance:

<Tabs>
<Tab title="Solo">
```ts corsair.ts
import { createCorsair } from "corsair";
import { cal } from "@corsair-dev/cal";

export const corsair = createCorsair({
    plugins: [
        cal(),
    ],
});
```
</Tab>
<Tab title="Multi-Tenant">
```ts corsair.ts
import { createCorsair } from "corsair";
import { cal } from "@corsair-dev/cal";

export const corsair = createCorsair({
    multiTenancy: true,
    plugins: [cal()],
    kek: process.env.CORSAIR_KEK!,
});
```
</Tab>
</Tabs>

Once configured, you can manage bookings:

<Tabs>
<Tab title="Solo">
```ts
// List all bookings
const bookings = await corsair.cal.api.bookings.list({});
```
</Tab>
<Tab title="Multi-Tenant">
```ts
const tenant = corsair.withTenant("user_id");

// List all bookings
const bookings = await tenant.cal.api.bookings.list({});
```
</Tab>
</Tabs>

## Authentication

### Supported Auth Types

The Cal.com plugin supports:
- **[`api_key`](/content/concepts/api-key)** (default) - Use a Cal.com API key

### Default Auth Type

If no `authType` is specified, the plugin defaults to **[`api_key`](/content/concepts/api-key)**.

### Configuring API Key Authentication

<Tabs>
<Tab title="Solo">
```bash
pnpm corsair setup --plugin=cal api_key=your-api-key
```

For webhook signature verification:

```bash
pnpm corsair setup --plugin=cal webhook_signature=your-webhook-secret
```
</Tab>
<Tab title="Multi-Tenant">
```bash
pnpm corsair setup --plugin=cal api_key=their-api-key --tenant=user_abc123
```

For webhook signature verification:

```bash
pnpm corsair setup --plugin=cal webhook_signature=your-webhook-secret
```
</Tab>
</Tabs>

See [API Key Authentication](/content/concepts/api-key) for full setup details.

## Options

| Option | Type | Description |
|--------|------|-------------|
| `authType` | `'api_key'` | Authentication method (defaults to `'api_key'`) |
| `key` | `string` | API key (optional, uses database if not provided) |
| `webhookSecret` | `string` | Webhook secret for `x-cal-signature-256` verification |
| `hooks` | `object` | Endpoint hooks for custom logic |
| `webhookHooks` | `object` | Webhook hooks for event handling |
| `errorHandlers` | `object` | Custom error handlers |
| `permissions` | `object` | Permission configuration for AI agent access |

### Hooks

```ts
cal({
    hooks: {
        bookings: {
            create: {
                before: async (ctx, input) => {
                    console.log("Creating booking:", input.eventTypeId);
                    return { ctx, input };
                },
            },
        },
    },
})
```

See [Hooks](/concepts/hooks) for complete documentation.

### Error Handling

The plugin includes built-in error handlers for common scenarios. For complete documentation, see the [Error Handlers reference](/plugins/cal/error-handlers).

## Usage

<Tabs>
<Tab title="Solo">
```ts
// Get a specific booking
const booking = await corsair.cal.api.bookings.get({
    uid: "booking-uid-123",
```

### Webhooks

Cal.com sends booking lifecycle events to your webhook endpoint:

<Tabs>
<Tab title="Solo">
```ts app/api/webhook/route.ts
import { processWebhook } from "corsair";
import { corsair } from "@/server/corsair";

export async function POST(request: Request) {
    const headers = Object.fromEntries(request.headers);
    const body = await request.json();

    const result = await processWebhook(corsair, headers, body);
    return result.response;
}
```
</Tab>
<Tab title="Multi-Tenant">
```ts app/api/webhook/route.ts
import { processWebhook } from "corsair";
import { corsair } from "@/server/corsair";

export async function POST(request: Request) {
    const headers = Object.fromEntries(request.headers);
    const body = await request.json();

    const tenantId = new URL(request.url).searchParams.get("tenantId") ?? undefined;
    const result = await processWebhook(corsair, headers, body, { tenantId });
    return result.response;
}
```
</Tab>
</Tabs>

See [Webhooks](/plugins/cal/webhooks) for all available events.

## Examples

### Example 1: Send Confirmation Email on New Booking

```ts corsair.ts
import { inngest } from "./inngest";

export const corsair = createCorsair({
    plugins: [
        cal({
            webhookHooks: {
                bookings: {
                    bookingCreated: {
                        after: async (ctx, result) => {
                            await inngest.send({ // [!code highlight]
                                name: "cal/booking-created", // [!code highlight]
                                data: { // [!code highlight]
                                    tenantId: ctx.tenantId, // [!code highlight]
                                    bookingUid: result.data.uid, // [!code highlight]
                                    attendeeName: result.data.attendees?.[0]?.name, // [!code highlight]
                                    attendeeEmail: result.data.attendees?.[0]?.email, // [!code highlight]
                                    startTime: result.data.startTime, // [!code highlight]
                                }, // [!code highlight]
                            }); // [!code highlight]
                        },
                    },
                },
            },
        }),
    ],
});
```

```ts inngest/functions.ts
export const sendBookingConfirmation = inngest.createFunction(
    { id: "send-booking-confirmation" },
    { event: "cal/booking-created" },
    async ({ event }) => {
        const tenant = corsair.withTenant(event.data.tenantId);

        await tenant.resend.api.emails.send({
            from: "bookings@example.com",
            to: event.data.attendeeEmail,
            subject: "Your booking is confirmed",
            text: `Hi ${event.data.attendeeName}, your meeting is scheduled for ${event.data.startTime}.`,
        });
    }
);
```

### Example 2: Add to CRM on Meeting Ended

```ts corsair.ts
export const corsair = createCorsair({
    plugins: [
        cal({
            webhookHooks: {
                bookings: {
                    meetingEnded: {
                        after: async (ctx, result) => {
                            await inngest.send({
                                name: "cal/meeting-ended",
                                data: {
                                    tenantId: ctx.tenantId,
                                    bookingUid: result.data.uid,
                                    attendeeEmail: result.data.attendees?.[0]?.email,
                                },
                            });
                        },
                    },
                },
            },
        }),
        hubspot(),
    ],
});
```

```ts inngest/functions.ts
export const updateCrmAfterMeeting = inngest.createFunction(
    { id: "update-crm-after-meeting" },
    { event: "cal/meeting-ended" },
    async ({ event }) => {
        const tenant = corsair.withTenant(event.data.tenantId);

        const contacts = await tenant.hubspot.api.contacts.search({
            filterGroups: [{
                filters: [{ propertyName: "email", operator: "EQ", value: event.data.attendeeEmail }], // [!code highlight]
            }],
        });

        if (contacts.data.results[0]) {
            await tenant.hubspot.api.contacts.update({
                contactId: contacts.data.results[0].id,
                properties: { last_meeting_date: new Date().toISOString() },
            });
        }
    }
);
```
