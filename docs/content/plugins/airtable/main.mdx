## Quick Start

Install the plugin:

```bash
pnpm install @corsair-dev/airtable
```

Add the Airtable plugin to your Corsair instance:

<Tabs>
<Tab title="Solo">
```ts corsair.ts
import { createCorsair } from "corsair";
import { airtable } from "@corsair-dev/airtable";

export const corsair = createCorsair({
    plugins: [
        airtable(),
    ],
});
```
</Tab>
<Tab title="Multi-Tenant">
```ts corsair.ts
import { createCorsair } from "corsair";
import { airtable } from "@corsair-dev/airtable";

export const corsair = createCorsair({
    multiTenancy: true,
    plugins: [airtable()],
    kek: process.env.CORSAIR_KEK!,
});
```
</Tab>
</Tabs>

Once configured, you can work with Airtable records:

<Tabs>
<Tab title="Solo">
```ts
// Search records in a table
const records = await corsair.airtable.api.records.search({
    baseId: "appXXXXXXXXXXXXXX",
    tableIdOrName: "Tasks",
    filterByFormula: "{Status} = 'In Progress'",
});
```
</Tab>
<Tab title="Multi-Tenant">
```ts
const tenant = corsair.withTenant("user_id");

// Search records in a table
const records = await tenant.airtable.api.records.search({
    baseId: "appXXXXXXXXXXXXXX",
    tableIdOrName: "Tasks",
    filterByFormula: "{Status} = 'In Progress'",
});
```
</Tab>
</Tabs>

## Authentication

### Supported Auth Types

The Airtable plugin supports:
- **[`api_key`](/content/concepts/api-key)** (default) - Use an Airtable personal access token

### Default Auth Type

If no `authType` is specified, the plugin defaults to **[`api_key`](/content/concepts/api-key)**.

### Configuring API Key Authentication

<Tabs>
<Tab title="Solo">
```bash
pnpm corsair setup --plugin=airtable api_key=your-personal-access-token
```

For webhook signature verification:

```bash
pnpm corsair setup --plugin=airtable webhook_signature=your-webhook-secret
```
</Tab>
<Tab title="Multi-Tenant">
```bash
pnpm corsair setup --plugin=airtable api_key=their-api-key --tenant=user_abc123
```

For webhook signature verification:

```bash
pnpm corsair setup --plugin=airtable webhook_signature=your-webhook-secret
```
</Tab>
</Tabs>

See [API Key Authentication](/content/concepts/api-key) for full setup details.

## Options

| Option | Type | Description |
|--------|------|-------------|
| `authType` | `'api_key'` | Authentication method (defaults to `'api_key'`) |
| `key` | `string` | Personal access token (optional, uses database if not provided) |
| `webhookSecret` | `string` | Webhook secret for `x-airtable-content-mac` verification |
| `hooks` | `object` | Endpoint hooks for custom logic |
| `webhookHooks` | `object` | Webhook hooks for event handling |
| `errorHandlers` | `object` | Custom error handlers |
| `permissions` | `object` | Permission configuration for AI agent access |

### Hooks

```ts
airtable({
    hooks: {
        records: {
            create: {
                before: async (ctx, input) => {
                    console.log("Creating record in:", input.tableIdOrName);
                    return { ctx, input };
                },
            },
        },
    },
})
```

See [Hooks](/concepts/hooks) for complete documentation.

### Error Handling

The plugin includes built-in error handlers for common scenarios. For complete documentation, see the [Error Handlers reference](/plugins/airtable/error-handlers).

### Webhooks

Airtable sends record change events to your webhook endpoint:

<Tabs>
<Tab title="Solo">
```ts app/api/webhook/route.ts
import { processWebhook } from "corsair";
import { corsair } from "@/server/corsair";

export async function POST(request: Request) {
    const headers = Object.fromEntries(request.headers);
    const body = await request.json();

    const result = await processWebhook(corsair, headers, body);
    return result.response;
}
```
</Tab>
<Tab title="Multi-Tenant">
```ts app/api/webhook/route.ts
import { processWebhook } from "corsair";
import { corsair } from "@/server/corsair";

export async function POST(request: Request) {
    const headers = Object.fromEntries(request.headers);
    const body = await request.json();

    const tenantId = new URL(request.url).searchParams.get("tenantId") ?? undefined;
    const result = await processWebhook(corsair, headers, body, { tenantId });
    return result.response;
}
```
</Tab>
</Tabs>

See [Webhooks](/plugins/airtable/webhooks) for all available events.

## Examples

### Example 1: Sync Form Submissions to Airtable

```ts inngest/functions.ts
export const syncFormSubmission = inngest.createFunction(
    { id: "sync-form-to-airtable" },
    { event: "app/form-submitted" },
    async ({ event }) => {
        const tenant = corsair.withTenant(event.data.tenantId);

        await tenant.airtable.api.records.create({
            baseId: process.env.AIRTABLE_BASE_ID!,
            tableIdOrName: "Leads",
            fields: {
                Name: event.data.name, // [!code highlight]
                Email: event.data.email, // [!code highlight]
                Message: event.data.message, // [!code highlight]
                Source: "Web Form",
                Submitted: new Date().toISOString(),
            },
        });
    }
);
```

### Example 2: React to Record Changes

```ts corsair.ts
export const corsair = createCorsair({
    plugins: [
        airtable({
            webhookHooks: {
                events: {
                    event: {
                        after: async (ctx, result) => {
                            await inngest.send({
                                name: "airtable/record-changed",
                                data: {
                                    tenantId: ctx.tenantId,
                                    tableId: result.data.tableId, // [!code highlight]
                                    recordId: result.data.recordId,
                                    changeType: result.data.changedFieldsById ? "update" : "create",
                                },
                            });
                        },
                    },
                },
            },
        }),
    ],
});
```

```ts inngest/functions.ts
export const processAirtableChange = inngest.createFunction(
    { id: "process-airtable-change" },
    { event: "airtable/record-changed" },
    async ({ event }) => {
        const tenant = corsair.withTenant(event.data.tenantId);

        const record = await tenant.airtable.api.records.get({
            baseId: process.env.AIRTABLE_BASE_ID!,
            tableIdOrName: event.data.tableId,
            recordId: event.data.recordId,
        });

        console.log("Changed record:", record.data.fields);
    }
);
```

### Example 3: Upsert Records on External Events

```ts inngest/functions.ts
export const upsertContact = inngest.createFunction(
    { id: "upsert-airtable-contact" },
    { event: "crm/contact-updated" },
    async ({ event }) => {
        const tenant = corsair.withTenant(event.data.tenantId);

        await tenant.airtable.api.records.createOrUpdate({
            baseId: process.env.AIRTABLE_BASE_ID!,
            tableIdOrName: "Contacts",
            performUpsert: {
                fieldsToMergeOn: ["Email"], // [!code highlight]
            },
            fields: {
                Email: event.data.email,
                Name: event.data.name,
                Company: event.data.company,
                LastUpdated: new Date().toISOString(),
            },
        });
    }
);
```
