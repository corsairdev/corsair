## Quick Start

Install the plugin:

```bash
pnpm install @corsair-dev/oura
```

Add the Oura plugin to your Corsair instance:

<Tabs>
<Tab title="Solo">
```ts corsair.ts
import { createCorsair } from "corsair";
import { oura } from "@corsair-dev/oura";

export const corsair = createCorsair({
    plugins: [
        oura(),
    ],
});
```
</Tab>
<Tab title="Multi-Tenant">
```ts corsair.ts
import { createCorsair } from "corsair";
import { oura } from "@corsair-dev/oura";

export const corsair = createCorsair({
    multiTenancy: true,
    plugins: [oura()],
    kek: process.env.CORSAIR_KEK!,
});
```
</Tab>
</Tabs>

Once configured, you can access Oura health data:

<Tabs>
<Tab title="Solo">
```ts
// Get sleep data for a date range
const sleep = await corsair.oura.api.summary.getSleep({
    start_date: "2024-01-01",
    end_date: "2024-01-07",
});
```
</Tab>
<Tab title="Multi-Tenant">
```ts
const tenant = corsair.withTenant("user_id");

// Get sleep data for a date range
const sleep = await tenant.oura.api.summary.getSleep({
    start_date: "2024-01-01",
    end_date: "2024-01-07",
});
```
</Tab>
</Tabs>

## Authentication

### Supported Auth Types

The Oura plugin supports:
- **`api_key`** (default) - Use an Oura personal access token

### Default Auth Type

If no `authType` is specified, the plugin defaults to **`api_key`**.

### Configuring API Key Authentication

<Tabs>
<Tab title="Solo">
```bash
pnpm corsair setup --oura api_key=your-access-token
```

For webhook signature verification:

```bash
pnpm corsair setup --oura webhook_signature=your-webhook-secret
```
</Tab>
<Tab title="Multi-Tenant">
```bash
pnpm corsair setup --oura api_key=their-access-token --tenant=user_abc123
```

For webhook signature verification:

```bash
pnpm corsair setup --oura webhook_signature=your-webhook-secret
```
</Tab>
</Tabs>

See [API Key Authentication](/concepts/api-key) for full setup details.

## Options

| Option | Type | Description |
|--------|------|-------------|
| `authType` | `'api_key'` | Authentication method (defaults to `'api_key'`) |
| `key` | `string` | Personal access token (optional, uses database if not provided) |
| `webhookSecret` | `string` | Webhook secret for `x-oura-signature` HMAC verification |
| `hooks` | `object` | Endpoint hooks for custom logic |
| `webhookHooks` | `object` | Webhook hooks for event handling |
| `errorHandlers` | `object` | Custom error handlers |
| `permissions` | `object` | Permission configuration for AI agent access |

### Hooks

```ts
oura({
    hooks: {
        summary: {
            getSleep: {
                before: async (ctx, input) => {
                    console.log("Fetching sleep data:", input.start_date);
                    return { ctx, input };
                },
            },
        },
    },
})
```

See [Hooks](/concepts/hooks) for complete documentation.

### Error Handling

The plugin includes built-in error handlers for common scenarios. For complete documentation, see the [Error Handlers reference](/plugins/oura/error-handlers).

### Webhooks

Oura sends daily health summary events to your webhook endpoint:

<Tabs>
<Tab title="Solo">
```ts app/api/webhook/route.ts
import { processWebhook } from "corsair";
import { corsair } from "@/server/corsair";

export async function POST(request: Request) {
    const headers = Object.fromEntries(request.headers);
    const body = await request.json();

    const result = await processWebhook(corsair, headers, body);
    return result.response;
}
```
</Tab>
<Tab title="Multi-Tenant">
```ts app/api/webhook/route.ts
import { processWebhook } from "corsair";
import { corsair } from "@/server/corsair";

export async function POST(request: Request) {
    const headers = Object.fromEntries(request.headers);
    const body = await request.json();

    const tenantId = new URL(request.url).searchParams.get("tenantId") ?? undefined;
    const result = await processWebhook(corsair, headers, body, { tenantId });
    return result.response;
}
```
</Tab>
</Tabs>

See [Webhooks](/plugins/oura/webhooks) for all available events.

## Examples

### Example 1: Daily Health Report

```ts inngest/functions.ts
export const dailyHealthReport = inngest.createFunction(
    { id: "oura-daily-report" },
    { cron: "0 8 * * *" },
    async () => {
        const today = new Date().toISOString().split("T")[0];
        const yesterday = new Date(Date.now() - 86400000).toISOString().split("T")[0];

        const [sleep, readiness, activity] = await Promise.all([
            corsair.oura.api.summary.getSleep({ start_date: yesterday, end_date: today }), // [!code highlight]
            corsair.oura.api.summary.getReadiness({ start_date: yesterday, end_date: today }), // [!code highlight]
            corsair.oura.api.summary.getActivity({ start_date: yesterday, end_date: today }), // [!code highlight]
        ]);

        console.log("Sleep score:", sleep.data[0]?.score);
        console.log("Readiness score:", readiness.data[0]?.score);
        console.log("Activity score:", activity.data[0]?.score);
    }
);
```

### Example 2: Alert on Low Readiness

```ts corsair.ts
export const corsair = createCorsair({
    plugins: [
        oura({
            webhookHooks: {
                summary: {
                    dailyReadiness: {
                        after: async (ctx, result) => {
                            await inngest.send({
                                name: "oura/readiness-updated",
                                data: {
                                    tenantId: ctx.tenantId,
                                    score: result.data.score, // [!code highlight]
                                    userId: result.data.user_id,
                                },
                            });
                        },
                    },
                },
            },
        }),
    ],
});
```

```ts inngest/functions.ts
export const alertOnLowReadiness = inngest.createFunction(
    { id: "alert-low-readiness" },
    { event: "oura/readiness-updated" },
    async ({ event }) => {
        if (event.data.score < 50) {
            // Send notification that user should rest
            console.log(`Low readiness (${event.data.score}) for user ${event.data.userId}`);
        }
    }
);
```
